  function formatAgo(iso) {
    if (!iso) return "";
    const t = new Date(iso).getTime();
    if (!isFinite(t)) return "";
    const diff = Date.now() - t;

    const min = Math.floor(diff / 60000);
    const hr = Math.floor(diff / 3600000);
    const day = Math.floor(diff / 86400000);

    if (day >= 60) return `${Math.floor(day/30)} months ago`;
    if (day >= 30) return `1 month ago`;
    if (day >= 14) return `${Math.floor(day/7)} weeks ago`;
    if (day >= 7) return `1 week ago`;
    if (day >= 2) return `${day} days ago`;
    if (day === 1) return `Yesterday`;
    if (hr >= 2) return `${hr} hours ago`;
    if (hr === 1) return `1 hour ago`;
    if (min >= 2) return `${min} min ago`;
    return `Just now`;
  }
