  // ---------- Local applications storage (shared with applications page) ----------
  const APPLICATIONS_KEY = "streamlatamApplicationsDemo";

  function safeJsonParse(str) { try { return JSON.parse(str); } catch { return null; } }

  function getApplications() {
    const data = safeJsonParse(localStorage.getItem(APPLICATIONS_KEY));
    return Array.isArray(data) ? data : [];
  }
  function setApplications(arr) {
    localStorage.setItem(APPLICATIONS_KEY, JSON.stringify(arr));
  }
  function isApplied(campaignId) {
    if (!campaignId) return false;
    return getApplications().some(a => a && a.campaignId === campaignId);
  }

  // ---------- Base data (for Fit chip) ----------
  function parseViewers(v) {
    if (!v) return 0;
    const cleaned = String(v).replace(/[^\d]/g, "");
    const n = parseInt(cleaned, 10);
    return isNaN(n) ? 0 : n;
  }

  const baseDefaults = { viewers: "145" };
  const storedBase = safeJsonParse(localStorage.getItem("streamerBaseData"));
  const base = (storedBase && typeof storedBase === "object") ? { ...baseDefaults, ...storedBase } : baseDefaults;
  const viewersNum = parseViewers(base.viewers);

  function computeFit(c) {
    const min = Number.isFinite(c.recommendedMin) ? c.recommendedMin : 0;
    const max = Number.isFinite(c.recommendedMax) ? c.recommendedMax : 999999;
    if (!viewersNum) return { label: "‚Äî", type: "warn" };

    if (viewersNum >= min && viewersNum <= max) return { label: "High", type: "ok" };

    // Soft fit (close-ish)
    const within = 30; // +-30 viewers threshold for "Medium"
    if ((viewersNum >= (min - within) && viewersNum < min) || (viewersNum > max && viewersNum <= (max + within))) {
      return { label: "Medium", type: "warn" };
    }
    return { label: "Low", type: "bad" };
  }

  // ---------- Query params ----------
  function getCampaignIdFromUrl() {
    const params = new URLSearchParams(window.location.search);
    return params.get("id") || "c1";
  }

    });

  // ---------- NEW: platform icon mapping ----------
  function platformIconHtml(platform){
    const p = String(platform || "").toLowerCase();
    if (p === "twitch") return `<i class="fa-brands fa-twitch"></i>`;
    if (p === "youtube") return `<i class="fa-brands fa-youtube"></i>`;
    if (p === "kick") return `<i class="fa-solid fa-bolt"></i>`;
    if (p === "trovo") return `<i class="fa-solid fa-tv"></i>`;
    return `<i class="fa-solid fa-tv"></i>`;
  }

  function setPublicProfileLink(c){

    openPublicProfile.href = `profile-public.html?campaignId=${encodeURIComponent(c.id)}`;
  }

  function setFitPillForCampaign(c) {
    if (!tbFitPill) return;
    const fit = computeFit(c);

    tbFitPill.classList.remove("ok", "warn", "bad");
    tbFitPill.classList.add(fit.type);

    const icon = fit.type === "ok" ? "fa-signal" : (fit.type === "warn" ? "fa-circle-info" : "fa-triangle-exclamation");
    tbFitPill.innerHTML = `<i class="fa-solid ${icon}"></i> Fit: ${fit.label}`;
  }

    // Header
    if (brandAvatar) brandAvatar.textContent = (c.brand || "B").trim().charAt(0).toUpperCase() || "B";
    if (titleMain) titleMain.textContent = c.title;
    if (subtitle) subtitle.textContent = `Brand: ${c.brand} ¬∑ Category: ${c.categoryLabel}`;

    if (feePill) feePill.innerHTML = `<i class="fa-solid fa-dollar-sign"></i> ${c.fee}`;
    if (typePill) typePill.innerHTML = `<i class="fa-solid fa-bullhorn"></i> ${c.typeLabel}`;
    if (durationPill) durationPill.innerHTML = `<i class="fa-solid fa-clock"></i> ${c.durationLabel}`;

    // Details
    if (expectText) expectText.textContent = c.expectsText;
    setList(expectList, c.expectsList);

    if (reqText) reqText.textContent = c.reqText;
    setList(reqList, c.reqList);

    // Side
    if (sideFee) sideFee.innerHTML = `${c.feeLine} <span></span>`;
    if (sideConditions) sideConditions.textContent = c.conditions;

    if (sideMeta && c.meta) {

        <div class="side-meta-item"><i class="fa-solid fa-language"></i> ${c.meta.language || "‚Äî"}</div>
        <div class="side-meta-item"><i class="fa-solid fa-calendar-week"></i> ${c.meta.timeline || c.duration || "‚Äî"}</div>
      `;
    }

    // Apply button state
    const already = isApplied(c.id);
    if (applyBtn) {
      applyBtn.disabled = already;
      applyBtn.style.opacity = already ? "0.75" : "1";
      applyBtn.style.cursor = already ? "not-allowed" : "pointer";

        ? `<i class="fa-solid fa-check"></i> Applied (demo)`
        : `<i class="fa-solid fa-paper-plane"></i> Apply to this campaign (demo)`;
    }

    if (savedBadge) savedBadge.classList.toggle("show", already);

    // NEW: keep brand view link synced with campaign
    setPublicProfileLink(c);
  }

  // Apply (demo): save locally as "In review"
  function applyToCampaign(c) {
    const list = getApplications();
    if (list.some(a => a && a.campaignId === c.id)) return;

    const now = new Date();
    list.unshift({
      campaignId: c.id,
      title: c.title,
      brand: c.brand,
      category: c.categoryLabel,   // IMPORTANT for applications.html
      fee: c.fee,
      status: "In review",
      createdAt: now.toISOString()
    });
    setApplications(list);

    showToast("Applied (demo)", "Saved locally. Check ‚ÄúMy applications‚Äù.");
    renderCampaign(c);
  }

  if (applyBtn) {
    applyBtn.addEventListener("click", () => {
      applyToCampaign(current);

      window.open(`profile-public.html?campaignId=${encodeURIComponent(current.id)}`, "_blank", "noopener");
    });
  }

    });
